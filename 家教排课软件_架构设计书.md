# 家教排课软件 - 架构设计书

## 项目代号：TutorSchedule / 家教通

---

## 一、项目概述

### 1.1 项目背景
家教排课软件是一款面向家教机构、独立教师及学生家长的课程管理工具，解决传统Excel排课效率低、冲突检测难、通知不及时等痛点。

### 1.2 核心功能
- **可视化排课**：周视图/日视图自由拖拽排课
- **冲突检测**：时间重叠、教室占用、教师冲突自动提醒
- **多方协同**：教师、学生、家长多端信息同步
- **课消管理**：课时记录、消耗统计、余额提醒
- **智能推荐**：基于历史记录推荐最优排课时间

### 1.3 目标平台
- Android 8.0+ (API 26+)
- 后续可扩展 iOS / Web / 小程序

### 3.2 架构模式说明

采用 **Clean Architecture + MVVM-MVI 混合架构**:

1. **表现层**: Jetpack Compose 构建UI，ViewModel持有UI State
2. **领域层**: 纯Kotlin模块，无Android依赖，包含业务用例和领域模型
3. **数据层**: Repository模式封装数据来源，对上层屏蔽本地/远程差异

```kotlin
    private val getWeeklySchedule: GetWeeklyScheduleUseCase,

    private val moveLesson: MoveLessonUseCase,
    private val deleteLesson: DeleteLessonUseCase,

    fun onIntent(intent: ScheduleIntent) {
        when (intent) {
            is ScheduleIntent.LoadWeek -> loadWeek(intent.week)
            is ScheduleIntent.CreateLesson -> createLesson(intent.lesson)
            is ScheduleIntent.MoveLesson -> moveLesson(intent.lessonId, intent.newSlot)
            is ScheduleIntent.DeleteLesson -> deleteLesson(intent.lessonId)
            is ScheduleIntent.CheckConflict -> checkConflict(intent.lesson)
            is ScheduleIntent.StartDrag -> updateDraggingState(true)
            is ScheduleIntent.EndDrag -> updateDraggingState(false)
        }
    }

    private fun loadWeek(week: WeekRange) {
        viewModelScope.launch {
            _uiState.value = ScheduleUiState.Loading
            getWeeklySchedule(week)
                .onSuccess { lessons ->
                    _uiState.value = ScheduleUiState.Success(
                        currentWeek = week,
                        lessons = lessons.map { it.toUiModel() }
                    )
                }
                .onFailure { error ->
                    _uiState.value = ScheduleUiState.Error(error.message ?: "加载失败")
                }
        }
    }
}
```

```kotlin
// 用例基类
abstract class UseCase<in P, out R> {
    abstract suspend operator fun invoke(params: P): Result<R>
}

// 获取周课表
class GetWeeklyScheduleUseCase @Inject constructor(
    private val lessonRepository: LessonRepository
) : UseCase<WeekRange, List<Lesson>>() {
    override suspend fun invoke(params: WeekRange): Result<List<Lesson>> {
        return lessonRepository.getLessonsForWeek(params)
    }
}

// 创建课程（含冲突检测）
class CreateLessonUseCase @Inject constructor(
    private val lessonRepository: LessonRepository,
    private val conflictChecker: ScheduleConflictChecker
) : UseCase<LessonDraft, Lesson>() {
    override suspend fun invoke(params: LessonDraft): Result<Lesson> {
        // 1. 冲突检测
        val conflicts = conflictChecker.check(params)
        if (conflicts.isNotEmpty()) {
            return Result.failure(ScheduleConflictException(conflicts))
        }
        // 2. 创建课程
        return lessonRepository.createLesson(params.toLesson())
    }
}

// 拖拽移动课程
class MoveLessonUseCase @Inject constructor(
    private val lessonRepository: LessonRepository,
    private val conflictChecker: ScheduleConflictChecker
) : UseCase<MoveLessonParams, Lesson>() {
    data class MoveLessonParams(
        val lessonId: LessonId,
        val newStartTime: LocalDateTime,
        val newEndTime: LocalDateTime,
        val newTeacherId: TeacherId? = null,
        val newRoomId: RoomId? = null
    )
    
    override suspend fun invoke(params: MoveLessonParams): Result<Lesson> {
        val original = lessonRepository.getLesson(params.lessonId)
            ?: return Result.failure(LessonNotFoundException())
        
        val draft = original.copy(
            schedule = ScheduleRule.OneTime(params.newStartTime, params.newEndTime),
            teacherId = params.newTeacherId ?: original.teacherId,
            roomId = params.newRoomId ?: original.roomId
        )
        
        val conflicts = conflictChecker.check(draft, excludeLessonId = params.lessonId)
        if (conflicts.isNotEmpty()) {
            return Result.failure(ScheduleConflictException(conflicts))
        }
        
        return lessonRepository.updateLesson(draft)
    }
}

// 检查排课冲突
class CheckScheduleConflictUseCase @Inject constructor(
    private val conflictChecker: ScheduleConflictChecker
) : UseCase<LessonDraft, List<Conflict>>() {
    override suspend fun invoke(params: LessonDraft): Result<List<Conflict>> {
        return Result.success(conflictChecker.check(params))
    }
}
```

#### 5.3.2 Repository 实现（Data层）

```kotlin
class LessonRepositoryImpl @Inject constructor(
    private val localDataSource: LessonLocalDataSource,
    private val remoteDataSource: LessonRemoteDataSource,
    private val networkMonitor: NetworkMonitor,
    private val syncManager: SyncManager
) : LessonRepository {

    // 本地优先策略：先读本地，后台同步远程
    override suspend fun getLessonsForWeek(week: WeekRange): Result<List<Lesson>> {
        // 1. 立即返回本地数据
        val localLessons = localDataSource.getLessonsForWeek(week)
        
        // 2. 如果有网络，后台同步
        if (networkMonitor.isOnline) {
            syncManager.syncWeek(week)
        }
        
        return Result.success(localLessons.map { it.toDomainModel() })
    }

    override fun observeLessonsForWeek(week: WeekRange): Flow<List<Lesson>> {
        return localDataSource.observeLessonsForWeek(week)
            .map { lessons -> lessons.map { it.toDomainModel() } }
            .flowOn(Dispatchers.IO)
    }

    override suspend fun createLesson(lesson: Lesson): Result<Lesson> {
        // 1. 本地写入
        localDataSource.insertLesson(lesson.toLocalModel())
        
        // 2. 标记待同步
        syncManager.markAsPending(lesson.id)
        
        // 3. 尝试立即同步（如果有网络）
        if (networkMonitor.isOnline) {
            syncManager.syncImmediately(lesson.id)
        }
        
        return Result.success(lesson)
    }
}
```

```kotlin
@Composable
fun DraggableLessonCard(
    lesson: LessonUiModel,
    slotHeight: Dp,
    onClick: () -> Unit,
    onDragEnd: (TimeSlot) -> Unit
) {
    val offsetY = remember { mutableFloatStateOf(0f) }
    val isDragging = remember { mutableStateOf(false) }
    
    val cardHeight = calculateCardHeight(lesson.startTime, lesson.endTime, slotHeight)
    val cardColor = TutorTheme.subjectColors[lesson.subject] ?: TutorTheme.md_colorPrimary
    
    Card(
        modifier = Modifier
            .offset { IntOffset(0, offsetY.floatValue.toInt()) }
            .height(cardHeight)
            .padding(2.dp)
            .pointerInput(Unit) {
                detectDragGesturesAfterLongPress(
                    onDragStart = { isDragging.value = true },
                    onDragEnd = { 
                        isDragging.value = false
                        // 计算落点，计算新时间段
                        val newSlot = calculateDropSlot(offsetY.floatValue)
                        onDragEnd(newSlot)
                    },
                    onDrag = { change, dragAmount ->
                        change.consume()
                        offsetY.floatValue += dragAmount.y
                    }
                )
            }
            .clickable(onClick = onClick),
        colors = CardDefaults.cardColors(
            containerColor = if (isDragging.value) cardColor.copy(alpha = 0.7f) else cardColor
        )
    ) {
        Column(modifier = Modifier.padding(8.dp)) {
            Text(
                text = lesson.subject,
                color = Color.White,
                style = TutorTheme.typography.labelMedium,
                fontWeight = FontWeight.Bold

                text = lesson.studentName,
                color = Color.White.copy(alpha = 0.9f),
```

```kotlin
enum class ConflictResolution {
    CLIENT_WINS,      // 客户端优先（离线编辑优先）
    SERVER_WINS,      // 服务端优先（管理员强制更新）
    LAST_WRITE_WINS,  // 时间戳优先
    MERGE             // 合并字段（部分更新可合并）
}

class SyncConflictResolver @Inject constructor(
    private val userPreferences: UserPreferences
) {
    fun resolve(
        local: LessonEntity,
        remote: LessonDto
    ): ConflictResolution {
        return when {
            // 本地编辑时间更新 → 本地优先
            local.updatedAt > remote.updatedAt.parse() -> ConflictResolution.CLIENT_WINS
            // 服务端标记为强制更新 → 服务端优先
            remote.forceUpdate -> ConflictResolution.SERVER_WINS
            // 默认：最后写入优先
            else -> ConflictResolution.LAST_WRITE_WINS
        }
    }
}
```
